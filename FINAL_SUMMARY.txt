  $ bash init.sh                     # 初始化项目
  $ source venv/bin/activate         # 激活虚拟环境

  $ docker-compose up -d             # 启动所有服务
  $ docker-compose ps                # 查看服务状态
  $ docker-compose logs -f           # 查看日志
